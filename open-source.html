Quick Start

1. Initialize

class MyApplication : Application() { override fun onCreate() { super.onCreate() NetworkLoggerProvider.logger.initialize(this) } }

2. Ktor Interceptor

val client = HttpClient { install(NetworkLoggerPlugin) { logger = NetworkLoggerProvider.logger filter = { req !req.url.toString().contains("analytics.com") } } }

3. WebSocket Logging

val ws = WebSocketLogger(NetworkLoggerProvider.logger) ws.logConnected(url) ws.logMessageReceived(url, text) ws.logMessageSent(url, text)
